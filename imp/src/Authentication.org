* Authentication Handler

#+NAME: auth_handler
#+BEGIN_SRC haskell
authCheck :: Maybe String -> ConnectionPool -> IO(Bool)
authCheck authSession pool = flip runSqlPersistMPool pool $ do
  case authSession of
    Nothing -> return False
    Just xs -> do
      ifExists <- selectFirst [SessionUserEmail ==. xs] []
      case ifExists of
        Nothing -> return False
        Just _  -> return True
